import TitleBar from "../components/TitleBar.vue"
import NavBar from "../components/NavBar.vue"
export default {
  name: 'PioneerCampUserInformation',
  data () {
    return {
      isOnLoad: true,
      specialColor: 1,
      pioneerCamp: 0,
      campStatus: false,
      search: '',
      userList: []
    }
  },
  created () {
    this.createUserInformation()
  },
  methods: {
    test () {
      alert(JSON.stringify(this.userList))
    },
    enterInfoPage(temp) {
      this.$store.commit('setAddicantIndex', temp)
      this.$router.push({
        name: 'AddicantInfomation'
      })
    },
    createUserInformation () {
      this.pioneerCamp = parseInt(sessionStorage['campIndex'])
      this.campStatus = (sessionStorage['campStatus'] === 'true')
      let signupUsers = []
      let attendUsers = []
      axios.get('http://localhost:8002/queryCampsignup/').then(response => {
        let campSignups = JSON.parse(response.data.data)
        for(let i in campSignups) {
          if(campSignups[i].fields.camp_id === this.pioneerCamp) {
            signupUsers[i] = campSignups[i].fields.user_id
          }
        }
      })
      axios.get('http://localhost:8002/queryCampattend/').then(response => {
        let campAttend = JSON.parse(response.data.data)
        for(let i in campAttend) {
          if(campAttend[i].fields.camp_id === this.pioneerCamp) {
            attendUsers[i] = campAttend[i].fields.user_id
          }
        }
      })
      for(let userIndex = 0; userIndex < signupUsers.length; userIndex++) {
        let userName = ''
        let userGender = ''
        let userPhoneNumber = ''
        let userCompany = ''
        let userPosition = ''
        let userStatus = false
        let classNumber = 0
        let userIndex = signinUser[userIndex]
        for(let i = 0; i < attendUsers.lenth; i++) {
          if(attendUsers[i] === userIndex) {
            userStatus = true
            break
          }
        }
        axios({
          url:'http://localhost:8002/selectUser/',
          method:'post',
          data:{
            id: signupUsers[userIndex]
          },
          transformRequest: [
            function (data) {
              return Qs.stringify(data);
            }
          ],
        }).then(response=>{
          let signinUser = JSON.parse(response.data.data)
          userName = signinUser[0].fields.user_name
          userGender = signinUser[0].fields.user_gender
          userPhoneNumber = signinUser[0].fields.user_telephone
        })
        axios({
          url:'http://localhost:8002/selectUserdata/',
          method:'post',
          data:{
            id: signupUsers[userIndex]
          },
          transformRequest: [
            function (data) {
              return Qs.stringify(data);
            }
          ],
        }).then(response=>{
          let signinUser = JSON.parse(response.data.data)
          userCompany = signinUser[0].fields.user_company
          userPosition = signinUser[0].fields.user_post
        })
        this.$set(this.userList, userIndex, {})
        this.$set(this.userList[userIndex], 'index', userIndex)
        this.$set(this.userList[userIndex], 'name', userName)
        this.$set(this.userList[userIndex], 'gender', userGender)
        this.$set(this.userList[userIndex], 'phoneNumber', userPhoneNumber)
        this.$set(this.userList[userIndex], 'company', userCompany)
        this.$set(this.userList[userIndex], 'position', userPosition)
        this.$set(this.userList[userIndex], 'status', userStatus)
        this.$set(this.userList[userIndex], 'class', classNumber)
      }
    },

      this.$router.push({
